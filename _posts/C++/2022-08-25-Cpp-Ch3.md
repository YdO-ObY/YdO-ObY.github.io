## 산술 연산자
```cpp
//  산술 연산자
//  Arithmetic Operator

#include <iostream>

using namespace std;

int main() {
    
    int x = 1;
    
    //부호를 바꿔주는 -연산자
    //단항 연산자
    int y = -x;
    //-1 출력
    
    x = 2;
    y = 4;
    
    //사칙연산
    //+ 더하기 연산자 
    int z = x + y;
    cout << z << endl;
    
    //- 뺼셈 연산자
    z = x - y;
    cout << z << endl;
    
    //* 곱셈 연산자
    z = x * y;
    cout << z << endl;
    
    // / 나눗셈 연산자
    z = x / y;
    cout << z << endl;
    
    //% 나머지 연산자
    z = x % y;
    cout << z << endl;
    
    x = 7; y = 4;
    //정수에서의 나머지 연산
    z = x / y;
    
    cout << "정수의 나눗셈 : " << z << endl;
    
    
    //실수에서의 나머지 연산
    //둘중 하나라도 실수면 실수로 나온다.
    cout << "실수의 나눗셈 : " << float(x) / y << endl;
    cout << "실수의 나눗셈 : " << x / float(y) << endl;
    cout << "실수의 나눗셈 : " << float(x) / float(y) << endl;
    
    //주의사항
    //음수를 나누는 경우
    //소수점 버림으로 처리
    cout << "음수의 나눗셈 : " << -5 / 2 << endl;
    
    //나머지 연산 음수의 경우
    //왼쪽의 부호에 따라 부호를 결정한다.
    
    
    //대입 연산자
    //x의 메모리의 정보를 a변수가 가르키고있는 메모리에 복사해 넣어라!
    int a = x;
    
    //축약형
    //코드가 짧아져서 좋다.
    a += y; //a = a + y;
    a -= y; //a = a - y;
    a *= y; //a = a * y;
    a /= y; //a = a / y;
    a %= y; //a = a % y;   
}
```

## 증감 연산자
```cpp
//  증가/감소 연산자
//  increment/decrement

#include <iostream>

using namespace std;

int add(int a , int b)
{
    return a + b;
}

int main() {
    //많이 쓰이고 쓰기 쉽고 실수가 많이발생한다. 중요하다.
    
    
    
    int x = 5;
//    int y = ++x;    //x에 1을 더한다.
    
    int y = --x;
    int z = x--;
    //앞뒤 둘다 가능
    
    cout << y << endl;
    cout << z << endl;
    //6출력
    
    
    
    int a = 6 , b = 6;
    
    cout << a << " " << b << endl;
    
//    cout << ++a << " " << --b << endl;
    //7 5 출력
    cout << a++ << " " << b-- << endl;
    //6 6 출력
    //증감이 안된다
    
    
    //앞에 붙는경우
    //증감 실행 후, 출력 실행
    //뒤에 붙는 경우
    //a b 출력 후, 증감 실행
    
    cout << a << " " << b << endl;
    //증감 발생
    
    cout << "---reset----" << endl;
    a = 6 , b = 6;
    
    cout << "1 : "<< a << " " << b << endl;
    //기본 출력
    cout << "2 : "<< ++a << " " << --b << endl;
    //증감후 출력
    cout << "3 : "<< a << " " << b << endl;
    //기본 출력
    cout << "4 : "<< a++ << " " << b-- << endl;
    //출력후 증감
    cout << "5 : "<< a << " " << b << endl;
    //기본 출력
 
    
    //부작용이 되는 경우
    //함수에 적용하는 경우
    //강의에서는4가 출력되지만, xcode에선 3이 출력된다.
    //컴파일러마다 다르게 적용이 된다.
    //정의하기 힘들다.
    int q = 1;
    int result = add(q, ++q);   //이렇게 사용하지 말아라. 인수에 증감 연산자를 사용하지말것;
    
    cout << result << endl;
    
    q = 1;
    int w = 2;
    
    result = add(q, ++w);       //이경우엔 4가 출력이됨;
    
    cout << result << endl;
    
    
    cout << "---reset----" << endl;
    
    x = 1;
    x = x++;            //이런식으로 사용하지말아라. 헷갈린다.
    
    cout << x << endl;
}

```

## 관계 연산자
```cpp
//  관계 연산자
//  Relational Operator

#include <iostream>

using namespace std;

int main() {
    //두 변수 대수비교에 사용하는 연산자이다.
    
//    while(true)
//    {
        int x, y;
        
        cin >> x >> y;
        
        cout << "Your input values are : " << x << " " << y << endl;
        
        //==연산자
        if(x == y)
            cout << "equal" << endl;
        
        //!= 연산자
        if(x != y)
            cout << "not equal" << endl;
        
        //>연산자
        if(x > y)
            cout << "x is greater than y" << endl;
        //<연산자
        if(x < y)
            cout << "x is less than y" << endl;

        //>=연산자
        if(x >= y)
            cout << "x is greater than y or equal to y" << endl;
        
        //<=연산자
        if(x <= y)
            cout << "x is less than y or equal to y" << endl;
//    }
    //반복문 처리로 연속 입력 가능

    //주의사항
    //실수의 비교에서 오류가ㅏ 발생한다.
    
    double d1(100 - 99.99); //0.001
    double d2(10 - 9.99);   //0.001
    //보기엔 같아보인다.
    
    if(d1 == d2)
        cout << "equal" << endl;
    else
    {
        cout << "not equal" << endl;    //여기가 출력
        if(d1 > d2)
            cout << "d1 > d2" << endl;  //여기가 출력
        else
            cout << "d1 < d2" << endl;
    }
//    if(d1 != d2)
//        cout << "not equal" << endl;
//
    //not equal 발생!!
    
    //수치해석학? 수치가 중요한곳에서 일하려면 이것도 공부해라
    cout << d1 - d2 << endl;
    //절댓값 함수 사용
    cout << std::abs(d1 - d2) << endl;
    //5.32907e-15 출력
    //이만큼 정도밖에 차이 안난다.
    
    //차이의 한계를 정해놓는다.
    //차이를 어떻게 정의를해야하는가?
    //정의 하는것은 그때그때 달라진다.
    const double epsilon = 1e-10;
    
    if(abs(d1 - d2) < epsilon)
        cout << "Approximately equal" << endl;      //여기 실행
    else
        cout << "not equal" << endl;
}
```
