## 함수와의 첫만남

//함수와의 첫 만남
```cpp
#include <iostream>

using namespace std;

//함수 네이밍은 정확하게 자세하게 적는것이 좋다.
int addTwoNumbers(int num_a, int num_b)
{
    int sum = num_a + num_b;
    
    //함수 명 앞의 int 와 출력 값의 형태가 같아야한다.
    //함수 출력 형태 :int
    //리턴값 sum의 형태 :int
    return sum;
}

//매개변수 num_a , num_b
int multiplyTwoNumbers(int num_a, int num_b)
{
    int sum = num_a * num_b;
    
    //함수 명 앞의 int 와 출력 값의 형태가 같아야한다.
    //함수 출력 형태 :int
    //리턴값,반환값 sum의 형태 :int
    return sum;
}


void printHelloWorld(void)
{
    cout << "Hello World!" << endl;
//    return 0; 해도되고 안해도 되고
    //return을 만나면 종료한다.
}

void printHello() //void 써도 안써도 ㄱㅊ
{
    cout << "Hello" << endl;
    
    //함수안에서 다른 함수 호출가능
    printHelloWorld();
    
    
    //함수안에 함수를 정의는 못한다.
}

int main()
{
    //함수    function
    //리턴값   return values
    //매개변수  parameters
    //인수    arguments
    int x = 0;
    
    cout << 1 + 2 << endl;
    
    //함수 호출부분
    //인수 1, 2
    cout << addTwoNumbers(1, 2) << endl;
    
    x = addTwoNumbers(100, 11);
    
    cout << addTwoNumbers(3, 4) << endl;

    
    cout << multiplyTwoNumbers(3, 4) << endl;
    //하나의 패턴이 여러번 나오는경우 함수를 만든다.
    //재사용이 가능하다.
    
    //함수도 메모리안에 저장이된다.
    
    
    
    printHello();

    
    
    printHelloWorld();
    
    
}
```
