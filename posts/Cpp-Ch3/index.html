//  산술 연산자
//  Arithmetic Operator

#include <iostream>

using namespace std;

int main() {
    
    int x = 1;
    
    //부호를 바꿔주는 -연산자
    //단항 연산자
    int y = -x;
    //-1 출력
    
    x = 2;
    y = 4;
    
    //사칙연산
    //+ 더하기 연산자 
    int z = x + y;
    cout << z << endl;
    
    //- 뺼셈 연산자
    z = x - y;
    cout << z << endl;
    
    //* 곱셈 연산자
    z = x * y;
    cout << z << endl;
    
    // / 나눗셈 연산자
    z = x / y;
    cout << z << endl;
    
    //% 나머지 연산자
    z = x % y;
    cout << z << endl;
    
    x = 7; y = 4;
    //정수에서의 나머지 연산
    z = x / y;
    
    cout << "정수의 나눗셈 : " << z << endl;
    
    
    //실수에서의 나머지 연산
    //둘중 하나라도 실수면 실수로 나온다.
    cout << "실수의 나눗셈 : " << float(x) / y << endl;
    cout << "실수의 나눗셈 : " << x / float(y) << endl;
    cout << "실수의 나눗셈 : " << float(x) / float(y) << endl;
    
    //주의사항
    //음수를 나누는 경우
    //소수점 버림으로 처리
    cout << "음수의 나눗셈 : " << -5 / 2 << endl;
    
    //나머지 연산 음수의 경우
    //왼쪽의 부호에 따라 부호를 결정한다.
    
    
    //대입 연산자
    //x의 메모리의 정보를 a변수가 가르키고있는 메모리에 복사해 넣어라!
    int a = x;
    
    //축약형
    //코드가 짧아져서 좋다.
    a += y; //a = a + y;
    a -= y; //a = a - y;
    a *= y; //a = a * y;
    a /= y; //a = a / y;
    a %= y; //a = a % y;   
}

//  증가/감소 연산자
//  increment/decrement

#include <iostream>

using namespace std;

int add(int a , int b)
{
    return a + b;
}

int main() {
    //많이 쓰이고 쓰기 쉽고 실수가 많이발생한다. 중요하다.
    
    
    
    int x = 5;
//    int y = ++x;    //x에 1을 더한다.
    
    int y = --x;
    int z = x--;
    //앞뒤 둘다 가능
    
    cout << y << endl;
    cout << z << endl;
    //6출력
    
    
    
    int a = 6 , b = 6;
    
    cout << a << " " << b << endl;
    
//    cout << ++a << " " << --b << endl;
    //7 5 출력
    cout << a++ << " " << b-- << endl;
    //6 6 출력
    //증감이 안된다
    
    
    //앞에 붙는경우
    //증감 실행 후, 출력 실행
    //뒤에 붙는 경우
    //a b 출력 후, 증감 실행
    
    cout << a << " " << b << endl;
    //증감 발생
    
    cout << "---reset----" << endl;
    a = 6 , b = 6;
    
    cout << "1 : "<< a << " " << b << endl;
    //기본 출력
    cout << "2 : "<< ++a << " " << --b << endl;
    //증감후 출력
    cout << "3 : "<< a << " " << b << endl;
    //기본 출력
    cout << "4 : "<< a++ << " " << b-- << endl;
    //출력후 증감
    cout << "5 : "<< a << " " << b << endl;
    //기본 출력
 
    
    //부작용이 되는 경우
    //함수에 적용하는 경우
    //강의에서는4가 출력되지만, xcode에선 3이 출력된다.
    //컴파일러마다 다르게 적용이 된다.
    //정의하기 힘들다.
    int q = 1;
    int result = add(q, ++q);   //이렇게 사용하지 말아라. 인수에 증감 연산자를 사용하지말것;
    
    cout << result << endl;
    
    q = 1;
    int w = 2;
    
    result = add(q, ++w);       //이경우엔 4가 출력이됨;
    
    cout << result << endl;
    
    
    cout << "---reset----" << endl;
    
    x = 1;
    x = x++;            //이런식으로 사용하지말아라. 헷갈린다.
    
    cout << x << endl;
}

//  sizeof, 쉼표연산자, 조건부 연산자
//  Comma Operator
//  Conditional Operator

#include <iostream>

using namespace std;

int getPrice(bool OnSale)
{
    if(OnSale)
        return 10;
    else
        return 100;;
}

int main() {
    
    //데이터 형의 크기를 알고 싶을때 사용
    //변수 타입사용가능
//    sizeof(float);
    
    //변수로도 사용가능
//    float a;
//    sizeof(a);
    
    //구조체 및 클래스도 사용가능
    //커스텀 자료형이다.
    
    
    //sizeof는 함수가 아니라 연산자이다.
    //변수일땐 괄호없이 사용가능;
    //cout << sizeof a ;

    
    //comma operator
    //주의 필요
    
    int x = 3;
    int y = 10;
    int z = (++x, ++y);     //++x; ++y; int z = y;      << for문에서 많이 사용한다.
    //콤마 앞 진행, 뒤 진행후 마지막 것을 대입한다.
    
    cout << x << " " << y << " " << z << endl;

    //선언의 콤마는 구분해주는 기호
    int a = 1 , b = 10;
    int c;
    c = a, b;
    //우선순위가 대입보다 낮아서 대입을 먼저 실행하고  콤마를 실행되어 (c = a) , b로 진행이되어 b는 아무런 진행이 안되고 a값이 c에 복사가 되어 a값을 출력한다.
    
    cout << c << endl;
    
    //for문외에는 잘 사용하지않는다. 보기불편하고 해석하기 불편하다.
    //코드는 보기 편하게 만드는게 제일 좋다.
    
    //conditional operator
    //arithmetric if
    //삼항 연산자
    
    bool onSale = true;
    
    int price;
    
    if(onSale)
        price = 10;
    else
        price = 100;
    
    cout << price << endl;
    
    //삼항연산자가 잘 쓰이는곳
    //price값을 const로 사용하고싶을때
    //연산이 간단할때 사용한다.
    const int price2 = (onSale == true) ? 10 : 100;
    
    cout << price2 << endl;
        

    //함수로 짤수도 있다.
    //코드가 간단하면 추천하지 않는다.
    //코드가 간단한 경우에는 코드가 늘어나고 보기 힘들어진다.
    const int price3 = getPrice(onSale);
    
}

//  관계 연산자
//  Relational Operator

#include <iostream>

using namespace std;

int main() {
    //두 변수 대수비교에 사용하는 연산자이다.
    
//    while(true)
//    {
        int x, y;
        
        cin >> x >> y;
        
        cout << "Your input values are : " << x << " " << y << endl;
        
        //==연산자
        if(x == y)
            cout << "equal" << endl;
        
        //!= 연산자
        if(x != y)
            cout << "not equal" << endl;
        
        //>연산자
        if(x > y)
            cout << "x is greater than y" << endl;
        //<연산자
        if(x < y)
            cout << "x is less than y" << endl;

        //>=연산자
        if(x >= y)
            cout << "x is greater than y or equal to y" << endl;
        
        //<=연산자
        if(x <= y)
            cout << "x is less than y or equal to y" << endl;
//    }
    //반복문 처리로 연속 입력 가능

    //주의사항
    //실수의 비교에서 오류가ㅏ 발생한다.
    
    double d1(100 - 99.99); //0.001
    double d2(10 - 9.99);   //0.001
    //보기엔 같아보인다.
    
    if(d1 == d2)
        cout << "equal" << endl;
    else
    {
        cout << "not equal" << endl;    //여기가 출력
        if(d1 > d2)
            cout << "d1 > d2" << endl;  //여기가 출력
        else
            cout << "d1 < d2" << endl;
    }
//    if(d1 != d2)
//        cout << "not equal" << endl;
//
    //not equal 발생!!
    
    //수치해석학? 수치가 중요한곳에서 일하려면 이것도 공부해라
    cout << d1 - d2 << endl;
    //절댓값 함수 사용
    cout << std::abs(d1 - d2) << endl;
    //5.32907e-15 출력
    //이만큼 정도밖에 차이 안난다.
    
    //차이의 한계를 정해놓는다.
    //차이를 어떻게 정의를해야하는가?
    //정의 하는것은 그때그때 달라진다.
    const double epsilon = 1e-10;
    
    if(abs(d1 - d2) < epsilon)
        cout << "Approximately equal" << endl;      //여기 실행
    else
        cout << "not equal" << endl;
}

//  논리 연산자
//  Logical Operator

#include <iostream>

using namespace std;

int main() {
    //bool값 연산 별로 어려운 내용은 아니다.
    //추상적이라 이해하기 힘들 수 있다.
    //하다보면 익숙해지니까 별걱정 말자.
    

    

    bool x = true;
    bool y = false;
    
    //logical not !
    //true -> fasle
    //false -> true
    cout << !x << endl;
    
    //logical and
    //&&
    //둘다 true일때만 true 반환
    cout << (x && y) << endl;
    
    //사용 예시 게임
    //타격 여부와 체력 조건에 의한 결과로 설명한다.
    bool hit = true;
    int health = 10;
    //if두개로 나눌수있지만, 더 복잡해므로 논리 연산자에 익숙해져야한다. 효율적이다.
    if(hit == true && health < 20) //if(hit && health < 20)도 가능 명확성이 조금 떨어져 보임
        cout << "die" << endl;
    else
        health -= 20;
    
    
    //logical or
    //둘중에 하나라도 true면 true 반환
    cout << (x || y) << endl;
    
    
    int a = 5;
    int b = 5;
    
    //equal 연산자의 잘못 사용 예시
    //우선순위로 인하여 원치않은 결과값이 나온다.
    if(!a == b)     //(!x) == y  //!(x == y)            //(x != y)
        cout << "a does not equal b" << endl;
    else
        cout << "a eqauls b" << endl;
    
    
    //예시
    int v = 1;
    
    if(v == 0 || v == 1)
        cout << "v is 0 or 1" << endl;
    //특정 아이템 사용시
    //and or로 한번 연습해봐라 자신만의 스토리를 이용하여
    
    
    //short circuit evaluation
    a = 2;
    b = 2;
    
    //생각으로는 왼쪽 연산하고 오른쪽 연산하고 두개의값을 비교한다. 라고 생각이 든다.
    //하지만, 실질적으로 and에서 왼쪽이 false가 되어버리면 계산을 안한다.
    //왜냐하면 and는 둘중 하나라도 false면 false가 반환이 되므로 왼쪽계산이 false라면 계산을 멈춘다.
    //or도 마찬가지로 왼쪽이 true면 계산을 멈추고 true를 반환한다.
    
    if(a == 1 && ( b++ == 2))
    {
        cout << "Test" << endl;
    }
    
    //a가 1인경우 3반환
    //a가 2인경우 2반환
    cout << b << endl;
    
    
    x = true;
    y = false;
    
    //De Morgan's Law
    
    // !(x && y)  !=   !x && !y;
    // !x || !y;
    
    // !(x || y) = !x && !y;
    
    
    
    //XOR  !=
    //false false   false
    //false true    true
    //true  false   true
    //true  true    false
    
    
    cout << boolalpha;
    cout << (x != y) << endl;
    cout << (x != !y) << endl;
    
    
    //연습 문제
    //(true && true) || false
    //(false && true) || true
    //(false && true) || false || true
    //(14 > 13 || 2 > 1) && (9 > 1)
    // !(2314123 > 2 || 123123 > 2387
    
    
    //and or 혼합
    
    bool b1 = true;
    bool b2 = false;
    bool b3 = false;
    
    
    bool r1 = b1 || b2 && b3;
    bool r2 = (b1 || b2) && b3;
    bool r3 = b1 || (b2 && b3);

//  비트단위 연산자
//  Bitwise Operator

#include <iostream>
//2진수로 바꿔주는 라이브러리
#include <bitset>

using namespace std;

int main() {
    //int float 타입단위 연산 사람이 하는 연산
    
    //비트단위
    //컴퓨터가 이진수로 저장하였을때 각자리를 계산하는 방식
    
    
    //1바이트가 최소단위
    
    //비트단위를 사용하며 더 빠른 경우가 있다.
    
    //비트단위
    //unsigned를 사용하는것이 대부분이다.
    
    unsigned int a = 3;
    
    //bitset<bit>(variable)
    cout << bitset<4>(a) << endl;
    
    
    
    unsigned int b = a << 1;
    

    // << left shift
    //cout 의 << 는 다른것이다. <오퍼레이터 오버로딩>
    cout << bitset<4>(b) << " " << b << endl;
    //0110 6
    
    //한칸더 이동
    b = b << 1;
    cout << bitset<4>(b) << " " << b << endl;
    //1100 12
    b = b << 1;
    cout << bitset<8>(b) << " " << b << endl;
    //00011000 24
    //규칙성이 보인다.
    
    
    cout << bitset<4>(a << 1) << " " << (a << 1) << endl;   //6  = 3 * 2^1
    cout << bitset<4>(a << 2) << " " << (a << 2) << endl;   //12 = 3 * 2^2
    cout << bitset<4>(a << 3) << " " << (a << 3) << endl;   //24 = 3 * 2^3
    cout << bitset<4>(a << 4) << " " << (a << 4) << endl;   //48 = 3 * 2^4
    
    //2의 제곱을 하고싶을경우 left shift를 사용하는것이 훨씬 빠르게 구현가능하다.

    
    //0011
    //0110
    //기능 : 이진수를 오른쪽으로 한칸 밀고 0으로 채워준다.
    //
    
    
    cout << "----right shift----" << endl;
    // >> right shift
    //cin 의 >> 는 다른것이다. <오퍼레이터 오버로딩>
    
    a = 1024;
    
    cout << bitset<16>(a) << endl;
    
    cout << bitset<16>(a >> 1) << " " << (a >> 1) << endl;   //512 = 1024 / 2^1
    cout << bitset<16>(a >> 2) << " " << (a >> 2) << endl;   //256 = 1024 / 2^2
    cout << bitset<16>(a >> 3) << " " << (a >> 3) << endl;   //128 = 1024 / 2^3
    cout << bitset<16>(a >> 4) << " " << (a >> 4) << endl;   //64  = 1024 / 2^4
    
//    0000010000000000
//    0000001000000000 512
//    0000000100000000 256
//    0000000010000000 128
//    0000000001000000 64
    
    
    cout << "----not bit----" << endl;
    // ~ not
    cout << bitset<16>(~a) << " " << (~a) << endl;
    //0과 1을 바꾸는 연산
    //1111101111111111 4294966271
    //1024 가 어떻게4294966271이 되었는가?
    
    cout << "----and bit----" << endl;
    // & and
    a = 0b1100;
    b = 0b0110;
    
    cout << a << " " << b << endl;
    //12 6
    cout << bitset<4>(a & b) << endl;
    //0100
    
    cout << "----or bit----" << endl;
    // | or
    cout << bitset<4>(a | b) << endl;
    //1110
    
    cout << "----xor bit----" << endl;
    // ^ xor
    cout << bitset<4>(a ^ b) << endl;
    //1010
    
    //비트연산자도 축약형으로 사용가능하다.
    //a &= b
    
    
    
    //Quiz
    //0110 >> 2
    //5 | 12
    //5 & 12
    //5 ^ 12
}

//  비트 플래그, 비트마스크사용법
//  Bit flags, Bit masks

#include <iostream>
#include <bitset>

using namespace std;

int main() {
    
    //비트 플래그
    
    //비트단위가 실전에서 어떻게 사용되는지 예시를 알려준다..
    
    //게임예시
    
    //아이템 착용 유무
    //4개의 장비 미착용중
    bool item1_flag = false;
    bool item2_flag = false;
    bool item3_flag = false;
    bool item4_flag = false;
    
    //착용!
    //1번 장비 작용
    item1_flag = true;
    
    //파괴!
    //1번 장비 파괴
    item1_flag = false;
    
    
    //특정 장비 착용 유무에 따른 이벤트 발생
    if(item3_flag == true)
    {
        //쏼라 쏼라~
    }

    
    //이렇게 적은경우엔 변수가 4개밖에 안되지만, 여러개가 있는경우 코드가 길어진다. 불편하다.
    //함수의 파라미터에 가짓수가 많아질수잇다.
    //invokeEvent(item1_flag,item2_flag,item3_flag,item4_flag ....)
    
    
    
    //유무 관련 데이터는 참 거짓 두가지만 가지므로 1바이트만 해도 8비트 이니까 8개 데이터 저장 가능하지않을까? 생각이 들수있다.
    
    
    unsigned char items_flag = 0;
    
    
    cout << bitset<8>(items_flag) << endl;
    //00000000
    
    //각 비트별 플래그 설정
    const unsigned char op0 = 1 << 0;
    const unsigned char op1 = 1 << 1;
    const unsigned char op2 = 1 << 2;
    const unsigned char op3 = 1 << 3;
    cout << bitset<8>(op0) << endl;
    cout << bitset<8>(op1) << endl;
    cout << bitset<8>(op2) << endl;
    cout << bitset<8>(op3) << endl;
    
    
    cout << "No item" << bitset<8>(items_flag) << endl;
    
    //item0 획득!
    items_flag |= op0;
    cout << bitset<8>(items_flag) << endl;
    //00000001
    
    //itme3 획득
    items_flag |= op3;
    cout << bitset<8>(items_flag) << endl;
    //00001001
    
    //item3 소멸
    items_flag &= ~op3;
    cout << bitset<8>(items_flag) << endl;
    
    
    //item1 여부 확인
    if(items_flag & op1)
    {
        cout << "있어요" << endl;
    }
    else
        cout << "없어요" << endl;
    
    //내 예상
//    if((items_flag & op0) == op0)
//        cout << "있어요" << endl;
    
    
    //아이템 동시 획득
    items_flag |= (op2 | op3);
    
    cout << bitset<8>(items_flag) << endl;
    
    
    //아이템2를 사용하요 아이템1 생성
    if((items_flag & op2) && !(items_flag & op1))
    {
        //두줄을 한줄로 줄일 수 있다.
        items_flag ^= op2;
        items_flag ^= op1;
        cout << bitset<8>(items_flag) << endl;
    }
    
    
    
    

    
    //비트 마스크
    //색상테이블 사용
    
    const unsigned int red_mask = 0xFF0000;
    const unsigned int green_mask = 0x00FF00;
    const unsigned int blue_mask = 0x0000FF;
    
//    cout << bitset<32>(red_mask) << endl;
//    cout << bitset<32>(green_mask) << endl;
//    cout << bitset<32>(blue_mask) << endl;
    
    unsigned int pixel_clr = 0xDAA520;
    
    cout << bitset<32>(pixel_clr) << endl;
    //00000000110110101010010100100000
    
    unsigned int red, green, blue;
    
    
    blue = pixel_clr & blue_mask;
    cout << bitset<8>(blue) << " " << (int)blue << endl;
    
    green = pixel_clr & green_mask;
    cout << bitset<16>(green) << " " << (int)green << endl;
    //1010010100000000
    //8칸 right shift 사용
    green >>= 8;
    cout << bitset<16>(green) << " " << (int)green << endl;
    
    red = pixel_clr & red_mask;
    cout << bitset<8>(red) << " " << (int)red << endl;

    red >>= 16;
    cout << bitset<8>(red) << " " << (int)red << endl;
    
    
    
}
