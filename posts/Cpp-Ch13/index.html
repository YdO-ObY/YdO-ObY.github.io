T getMax(T x, T y)
{
    return (x > y) ? x : y;
}

int main() {
    //이번시간부터 템플릿에 대해 알아볼것이다.
    
    //프로그래머의 단순작업을 줄여주기 위한 방법들이 여러가지가 존재한다.
    //그중 하나의기술인 템플릿에 대해 알아보자

    //템플릿을 사용하면 수월해진다.

    cout << getMax(1, 2) << endl;
    cout << getMax(1.2, 2.5) << endl;
    cout << getMax(1.2f, 2.7f) << endl;
    cout << getMax('a', 'c') << endl;
    //컴파일러가 알아서 자료형을 맞춰서 실행한다.
    
    //사용자 정의 타입도 사용가능하다.
    //class 타입의 함수 사용
    cout << getMax(Cents(5), Cents(9)) << endl;
    //< , << 오버로딩이 필요하다.
    //템플릿을 사용할때 사용하기위해 필요한 기능들이 구현이 되어있어야한다.
    
    //instantiation
    //객체 인스턴스 - 객체가 데이터를 가지고 존재하는것
    //템플릿 인스턴스 - 자료형이 정해져있지않고 함수를 호출할때 자료형이 정해질때
    //~~형으로 인스턴스화 되었다라고 말할수있다.
    
    return 0;
}

#pragma once

#include <assert.h>
#include <iostream>


//변경사항 - int말고 다른 자료형도 들어가게 만들어라

template<typename T, unsigned int T_SIZE>
//array의 크기를 지정해줄 non-type parameter를 만들어주는 방법

class Myarray
{
private:
//    int m_length;   길이는 T_SIZE가 컴파일타임에 결정이 되므로 필요없다.
    T *m_data;    //T type을 담기 위한 포인터
    
public:
    
    Myarray()
    {
        m_data = new T [T_SIZE];    //동적할당 크기 지정
    }
    
    ~Myarray()
    {
        reset();
    }
    
    void reset()
    {
        delete [] m_data;
        m_data = nullptr;
    }
    
    //int & operator[] (int index)
    T & operator[] (int index)    //반환형 수정
    {
        assert(index >= 0 && index < T_SIZE);
        return m_data[index];
    }
    
    int getLength()
    {
        return  T_SIZE;
    }
    
    //Non-type parameter의 경우
    //explicit instantiation로 인해 구현부를 나누게 되어버리면
    //크기가 각 경우에 따라 추가가 되어야하므로 헤더에 구현하는것이 좋다.
    void print()
    {
        for (int i = 0; i < T_SIZE; ++i)
            std::cout << m_data[i] << " ";
        
        std::cout << std::endl;
    }
    
};

#pragma once

template<class T>
class Storage8
{
private:
    T m_array[8];
    
public:
    void set(int idx, const T& value)
    {
        m_array[idx] = value;
    }
    
    const T& get(int idx)
    {
        return m_array[idx];
    }
};

template<>
class Storage8<bool>
{
private:
    unsigned char m_data;
    
public:
    Storage8() : m_data(0)
    {}
    
    void set(int idx, bool value)
    {
        unsigned char mask = 1 << idx;
        
        if(value)
            m_data |= mask;
        else
            m_data &= ~mask;
    }
    
    bool get(int idx)
    {
        unsigned char mask = 1 << idx;
        return (m_data & mask) != 0;
    }
};

using namespace std;

template<class T, int size>
//파라미터가 두가지인 템플릿
class StaticArray_BASE
{
private:
    T m_array[size];
    
public:
    T * getArray() {return m_array;}
    
    T& operator[](int idx)

// 포인터에 대한 템플릿 특수화

#include <iostream>

using namespace std;

template<class T>
class A
{
private:
    T m_value;
    
public:
    A(const T & input) : m_value(input)
    {}
    
    void print()
    {
        cout << m_value << endl;
        //포인터 값의 경우 주소값 출력
    }
};


//포인터에 대한 템플릿
template<class T>
class A<T*>
{
private:
    T* m_value;
    
public:
    A(T * input) : m_value(input)
    {}
    
    void print()
    {
        cout << *m_value << endl;
        //포인터의 경우 값을 출력
    }
};

int main() {
    //이번에는 파라미터가 포인터의 경우 템플릿 특수화에 대해 알아보자
        
    
    A<int> a_int(123);
    a_int.print();
    
    int temp = 456;
    A<int*> a_int_ptr(&temp);
    a_int_ptr.print();
    
    double temp_d = 3.141592;
    A<double*> a_double_ptr(&temp_d);
    a_double_ptr.print();
    
    return 0;
}

// 멤버 함수를 한번 더 템플릿화 하기
// Templatize

#include <iostream>

using namespace std;


//A class templatize
template<class T>
class A
{
private:
    T m_value;
    
public:
    A(const T & input) : m_value(input)
    {}
    
    void print()
    {
        cout << m_value << endl;
        //포인터 값의 경우 주소값 출력
    }
    
    //멤버 함수 템플릿
    template<typename  TT>
    void doSomething(const TT & input)
    {
        cout << typeid(T).name() << ", " << typeid(TT).name() << endl;
    }
};
int main() {
    //이번에는 클래스가 템플릿화되어있는 경우에 멤버함수에 추가적으로 멤버함수를 템플릿화 하는 방법에 대해 알아보자
    
    
    A<int> a_int(123);
    
    a_int.print();
    
    a_int.doSomething<float>(123);
    //T = int , TT = float
    
    a_int.doSomething(123.4);
    //파라미터로 구분이 되는경우 템플릿 파라미터 필요없다.
    //T = int , TT = double
    return 0;
}
